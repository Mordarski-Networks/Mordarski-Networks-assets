! Blocks TikTok
www.google.*##.g:has(a[href*="tiktok.com"])
www.google.*##a[href*="tiktok.com"]:upward(1)
duckduckgo.com##ol.react-results--main a[href*="tiktok.com"]:upward(article)
duckduckgo.com##.tile-wrap a[href*="tiktok.com"]:upward(.tile)
||tiktok.com^$all
